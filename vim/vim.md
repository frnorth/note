1. echo "set tabstop=4" >> /etc/vimrc  
set tabstop=4          //tab键为4个空格
set softtabstop=4   //按退格键一次能删除4个空格
set shiftwidth=4      //当行之间交错使用4个空格
set autoindent        //自动对齐
set cindent             //定制C语言风格的缩进
set number            //显示行号
set showmatch      //设置为匹配模式
set mouse=a         //使用鼠标，用滚轮时，文本可以上下移动
